<script>
  const menuToggle = document.getElementById('menu-toggle');
  const mainNav = document.getElementById('main-nav');
  const ctaHeader = document.querySelector('.cta-header');

  function adjustMenu() {
    if (window.innerWidth <= 768) {
      menuToggle.style.display = 'block';
      mainNav.style.display = 'none';
      ctaHeader.style.display = 'none';
    } else {
      menuToggle.style.display = 'none';
      mainNav.style.display = 'flex';
      ctaHeader.style.display = 'block';
    }
  }

  menuToggle.addEventListener('click', () => {
    const visible = mainNav.style.display === 'flex';
    mainNav.style.display = visible ? 'none' : 'flex';
    ctaHeader.style.display = visible ? 'none' : 'block';
  });

  window.addEventListener('resize', adjustMenu);
  adjustMenu();
</script><div id="header-container"></div>
